function gerarCertidaoJusticaFederal(requerenteId) { // Mostrar loading const btn = event.target; const originalText = btn.innerHTML; btn.innerHTML = ' Gerando...'; btn.disabled = true; fetch(`/gerar-certidao-justica-federal/${requerenteId}/`, { method: 'POST', headers: { 'X-CSRFToken': document.querySelector('[name=csrfmiddlewaretoken]').value, 'Content-Type': 'application/json' } }) .then(response => response.json()) .then(data => { if (data.success) { // Mostrar sucesso showAlert('success', data.message); // Fazer download automático if (data.url_download) { const link = document.createElement('a'); link.href = data.url_download; link.download = data.nome_arquivo; document.body.appendChild(link); link.click(); document.body.removeChild(link); } } else { showAlert('error', data.message); } }) .catch(error => { showAlert('error', 'Erro na comunicação com o servidor'); console.error('Erro:', error); }) .finally(() => { // Restaurar botão btn.innerHTML = originalText; btn.disabled = false; }); }
